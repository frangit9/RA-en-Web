loader.load(
  './baller_roblox.glb', // Ruta del baller
  (gltf) => {
    const model = gltf.scene;

    //Como el modelo del baller venía en condiciones raras, vamos a resolver problemas para rotar
    //por medio de un envoltorio, el cual vamos a manipular a voluntad

    // Creamos la caja en base al baller
    const box = new THREE.Box3().setFromObject(model);

    // Obtenemos el centro de la caja
    const center = new THREE.Vector3();
    box.getCenter(center);

    // Movemos la geometría del modelo para que quede su centro en [0,0,0]
    model.position.sub(center);


    //Ahora creamos ese contenedor que envuelve al baller

    // Lo creamos
    const wrapper = new THREE.Group();

    // Aplicamos las transformaciones al contenedor, no al baller
    wrapper.scale.set(0.4, 0.4, 0.4);
    wrapper.position.y = 0.1;
    //wrapper.rotation.y = Math.PI; 

    // Añadimos el baller centrado al wrapper
    wrapper.add(model);

    //y agregamos el wrapper al anclaje, no el balelr
    anchor.group.add(wrapper);
  },
  (error) => {
    console.error('Error al cargar el modelo GLB:', error);
  }
);
